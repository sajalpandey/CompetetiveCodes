class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int temp,n=matrix.size();
        for(int i=0;i<n;i++)
        {
            for(int j=i;j<n-i-1;j++)
            {
                temp=matrix[i][j];
                matrix[i][j]=matrix[n-j-1][i];
                matrix[n-j-1][i]=matrix[n-1-i][n-1-j];
                matrix[n-1-i][n-j-1]=matrix[j][n-1-i];
                matrix[j][n-1-i]=temp;
            }
        }
    }
};
