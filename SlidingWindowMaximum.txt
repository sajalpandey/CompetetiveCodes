class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        vector<int> ans;
        deque<int> dq;
        for(int i=0;i<nums.size();i++)
        {
            if(!dq.empty() && dq.front()<=i-k)
                dq.pop_front();
            while(!dq.empty() && nums[i]>nums[dq.back()])
                dq.pop_back();
            dq.emplace_back(i);
            if(i>=k-1)
                ans.emplace_back(nums[dq.front()]);
        }
        return ans;
    }
};
